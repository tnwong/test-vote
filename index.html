async function verify() { const inputField = document.getElementById('userInput'); const input = inputField.value.trim(); const msg = document.getElementById('msg'); // 網址必須單獨一行，不能跟在 // 後面 const api_url = "https://script.google.com/macros/s/AKfycbzFxnRwSyqjIsDa8itFRDQXjIqvmAMp0Unp7LweMOFnItUiz780sQINNsce9AgS9RPS/exec"; if (!input) return; msg.innerText = "驗證中..."; msg.style.color = "#aaa"; try { const finalUrl = api_url + "?key=" + encodeURIComponent(input); // 發送連線請求 const response = await fetch(finalUrl); const result = await response.json(); if (result.status === "success") { msg.style.color = "#00e676"; msg.innerText = "驗證成功！正在跳轉..."; setTimeout(() => { // 跳轉至 SurveyCake window.top.location.href = atob(result.url); }, 800); } else { msg.style.color = "#ff5252"; msg.innerText = "授權碼無效"; inputField.value = ""; } } catch (err) { console.error("錯誤:", err); msg.innerText = "系統連線失敗，請檢查權限"; } }
