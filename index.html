async function verify() { const input = document.getElementById('userInput').value.trim(); const msg = document.getElementById('msg'); // 務必替換成你在步驟 2 拿到的「最新」網址 const api_url = "https://script.google.com/macros/s/AKfycbyCyrG_unfAgjiPkqvmhZUPr9UkwpT6fyatQ4X93ZttzU09Y_6d8eI7U6ragF5CwSg/exec"; if (!input) return; msg.innerText = "驗證中..."; try { // 使用 GET 請求，並手動補上參數 const finalUrl = api_url + "?key=" + encodeURIComponent(input); const response = await fetch(finalUrl, { method: 'GET', mode: 'cors', // 明確要求跨網域模式 }); const result = await response.json(); if (result.status === "success") { msg.style.color = "#00e676"; msg.innerText = "驗證成功！正在跳轉..."; setTimeout(() => { window.top.location.href = atob(result.url); }, 800); } else { msg.style.color = "#ff5252"; msg.innerText = "授權碼無效"; } } catch (err) { console.error("詳細錯誤訊息:", err); msg.innerText = "系統連線錯誤，請確認網路或聯繫管理員"; } }
