async function checkAuth() { const pw = document.getElementById('pw').value.trim(); const btn = document.querySelector('#auth-section button'); const msg = document.getElementById('msg'); if(!pw || btn.disabled) return; // 鎖定按鈕防止重複點擊 btn.disabled = true; btn.innerText = "⏳ 系統驗證中..."; msg.innerText = "正在檢查投票資格..."; msg.style.color = "#1a73e8"; try { const resp = await fetch(`${apiUrl}?key=${encodeURIComponent(pw)}`); const data = await resp.json(); if (data.auth) { // 驗證成功，跳轉到投票畫面 currentKey = pw; document.getElementById('auth-section').style.display = 'none'; document.getElementById('vote-section').style.display = 'block'; } else { // 💡 關鍵：顯示從 GAS 傳回的具體錯誤訊息 (如「已完成投票」) msg.innerText = data.msg; msg.style.color = "red"; btn.disabled = false; btn.innerText = "進入投票系統"; } } catch (e) { msg.innerText = "⚠️ 系統繁忙或連線超時，請檢查 GAS 權限"; btn.disabled = false; btn.innerText = "進入投票系統"; } }
